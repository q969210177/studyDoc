<script lang='ts' setup>
<template>
  <div class="goodsEbookList">
    <BKTable class="p-x-10px" @register="register" :columns="columns">
      <template #toolbar_buttons>
        <ElButton type="primary" size="small" @click="handleOpenDraw">
          新增
        </ElButton>
      </template>
      <template #action="scope">
        <BKTableBtnAction
          :actions="[
            {
              label: `删除`,
              onClick: () => onDelete(scope.row),
            },
          ]"
        />
      </template>
    </BKTable>
    
  </div>
</template>
import { useBKTable } from '@/components/BKTable/hooks/useBKTable'
import { IBKTableProps } from '@/components/BKTable'
// 插槽可以直接在数据这里定义 也可以 写一个字符串 在模板里面定义
const columns:IBKTableProps["columns"] = [ { field: 'relation_id', title: '关联ID',slots:{default:(scope:any)=>scope.row.xxx} },
    { field: 'related_type_txt', title: '关联范围',slots:{default:'related_type_txt'} },
    { field: 'goods_id', title: '商品ID' },
    { field: 'resource_name', title: '资源名称' },
    { field: 'bkdh', title: '博看代号' },
    { field: 'resource_from_txt', title: '资源来源' },
    { field: 'resource_type_txt', title: '资源类型' },
    { field: 'resource_sub_type_txt', title: '资源子类' },
    { field: 'start_date', title: '生效时间' },
    { field: 'cn', title: '生效期集' },

BKTable Props 类型定义 ​

属性名类型默认值说明
columnsArray as PropType(IBKTableProps['columns'])[]表头以及数据字段,各类插槽的定义
pagerConfigObject as PropType(PagerConfig){}分页设置（一般不用设置）
apiFunction as PropType((...arg: any[]) => Promise(any))undefined请求的方法 返回值为{total: 0,
apiParamsObject as PropType(Record<string, any>)nullapi的固定参数
keyFieldString as PropType(string)''
showCheckboxBoolean as PropType(boolean)false是否展示多选
showRadioBoolean as PropType(boolean)false是否展示单选
showIndexBoolean as PropType(boolean)true是否展示序号
autoHeightBoolean as PropType(boolean)true自动计算高度
showPagerBoolean as PropType(boolean)true是否展示分页(无分页场景也不会有分页参数)
immediateBoolean as PropType(boolean)true初始是否调用api
rowConfigObject as PropType(VxeTablePropTypes.RowConfig){}table的行设置
editConfigObject as PropType(VxeTablePropTypes.EditConfig){}编辑行的设置(例如单击变成编辑表格还是双击变成编辑表格等等 参考vxetable)
checkboxConfigObject as PropType(VxeTablePropTypes.CheckboxConfig){}多选单元设置(参考vxetable)
toolbarConfigObject as PropType(VxeToolbarProps){}一般不用管
actionWidthNumber as PropType(number)160操作栏的宽度
actionPropsObject as PropType(VxeTableDefines.ColumnOptions){}(参考vxetable)
rowClassNameString as PropType(VxeTablePropTypes.RowClassName)''行的class
showActionBoolean as PropType(boolean)true是否展示操作栏(是有一个action的插槽,否则没有)
其他VxeGridProps的所以参数会透传进来VxeGrid的props

TableEventMethods 类型定义 ​

方法名类型说明
setTableProps(props: IBKTableProps) => void | PromiseLike(void)设置表格属性
clearRowKeys() => void清空选中行
removeCheckboxRow() => Promise(void)清空选中行
toggleCheckboxRow(row: any) => void切换某一行的选择状态
setAllCheckboxRow(checked: boolean) => void设置所有行的选中状态
setCheckboxRow(rows: any, checked: boolean) => void设置行为选中状态，第二个参数为选中与否
setRadioRow(rows: any, checked: boolean) => void设置某一行为选中状态
checkAllRows(isFull?: boolean) => any[] | Promise(any[])获取选中的行数据
getColumnByField(field: string) => VxeTableDefines.ColumnInfo | Promise(VxeTableDefines.ColumnInfo) | null获取列信息
hideColumn(name: string) => void隐藏列
showColumn(name: string) => void显示列
addRow(args: any) => Promise({ row: any, rows: any[] })新增行
addRowsAt(record: any | any[], row?: any) => Promise({ row: any, rows: any[] })新增多行到指定位置
editRowEvent(row: any, fieldOrColumn?: string | VxeTableDefines.ColumnInfo) => Promise(any)激活编辑行
setTreeExpand(rows: any, checked: boolean) => Promise(any)用于 tree-config，设置展开树形节点，二个参数设置这一行展开与否
editRowCellEvent(row: any, fieldOrColumn: string | VxeTableDefines.ColumnInfo) => Promise(any)选择编辑单元格
clearEdit() => Promise(any)取消编辑状态
cancelRowEvent(row: any) => void取消当前编辑状态并还原数据
getActionRows() => { insertRecords: any, removeRecords: any, updateRecords: any } | Promise({ insertRecords: any, removeRecords: any, updateRecords: any })获取表格数据集（获取插入、删除、更改的数据）
validate() => Promise(boolean)表格验证
fullValidate() => Promise(boolean)表格完整验证
getTableData() => { fullData: any[], visibleData: any[], tableData: any[], footerData: any[][] } | Promise({ fullData: any[], visibleData: any[], tableData: any[], footerData: any[][] })获取表格数据
getFullData() => any[] | Promise(any[])获取表格数据 -> 全部数据
getVisibleData() => any[] | Promise(any[])获取表格数据 -> 显示数据
getFooterData() => any[] | Promise(any[])获取表格数据 -> 底部数据
isEditByRow(row: any) => boolean | Promise(boolean)判断 row 是否为编辑状态
removeRow(row?: any) => void删除行
updateStatus(row: any) => void更改某一行的状态
loadData(data: any) => void重载表格数据
reloadData(data: any) => void重载表格数据并清除所有状态
getUpdateRecords() => any[] | Promise(any[])获取表格数据编辑
setCurrentRow(row: any) => Promise(void)设置某行高亮
getInsertRecords() => any[] | Promise(any[])获取插入数据
getData(rowIndex: number) => any获取某一行数据
clearData() => void清空数据
getCurrentRecordRow() => any获取高亮行
getTableColumn() => { collectColumn: any, fullColumn: any, visibleColumn: any, tableColumn: any } | Promise({ collectColumn: any, fullColumn: any, visibleColumn: any, tableColumn: any })获取当前表格的列
clearCurrentRow() => void清空高亮
